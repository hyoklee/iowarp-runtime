project(chimaera)

set(CMAKE_CXX_STANDARD 17)

#------------------------------------------------------------------------------
# Build Tests
#------------------------------------------------------------------------------

        ${TEST_MAIN}/main_mpi.cc
        test_finalize.cc
        test_ipc.cc

)

        Catch2::Catch2 MPI::MPI_CXX OpenMP::OpenMP_CXX
